# Самостоятельная работа №1
### Дайте подробный комментарий для кода, написанного ниже. Комментарий нужен для каждой строчки кода, нужно описать что она делает. Не забудьте, что функции комментируются по-особенному.

```python
from datetime import datetime  # импортируется класс datetime для работы с датой и временем
from math import sqrt  # импортируется функция sqrt для вычисления квадратного корня


def main(**kwargs):
    """
    Возвращает квадратный корень суммы квадратов двух элементов кортежей

    :param kwargs: принимаемые кортежи, где ключи являются именами, а значения списками координат
    :rtype: float
    :return: печатает result в консоли
    """

    for key in kwargs.items():   # итерация по элементам словаря kwargs, где key - это кортеж
        result = sqrt(key[1][0] ** 2 + key[1][1] ** 2)  # вычисление квадратного корня суммы квадратов элементов
        print(result)  # вывод результата на консоль


if __name__ == '__main__':  # проверка, является ли модуль основным
    start_time = datetime.now()  # фиксация начала времени

    # вызов функции main с передачей словаря, содержащего списки координат
    main(
        one=[10, 3],  # первый кортеж
        two=[5, 4],  # втрой кортеж
        three=[15, 13],  # третий кортеж
        four=[93, 53],  # четвертый кортеж
        five=[133, 15]  # пятый кортеж
    )

    # подсчёт сколько времени заняло выполнение программы. Сейчас - начало времени
    time_costs = datetime.now() - start_time

    # Выводит в консоль сколько времени выполнялась программа
    print(f"Время выполнения программы - {time_costs}")

```

Основная функция `main(kwargs):` принимает словарь kwargs с координатами и вычисляет квадратные корни суммы квадратов этих координат.

 Цикл for проходит по элементам словаря и выполняет  вычисления, результат выводится на консоль.

```python
def calculate_average(*args):
    if not args: 
        return 0  
    return sum(args) / len(args)  # считаем среднее арифметическое

def main():
    
    numbers = input("Введите числа через пробел: ")
    # преобразуем введенные строки в числа
    num_list = [float(num) for num in numbers.split()]

    average = calculate_average(*num_list)  # вызываем функцию с распаковкой аргументов
    print(f"Среднее арифметическое: {average}")

if __name__ == '__main__':
    main()
```

```python
from math import sqrt


def triangle_area(a, b, c):
    # проверка на возможность существования треугольника
    if a + b <= c or a + c <= b or b + c <= a:
        raise ValueError("Треугольник с такими сторонами не существует.")

    p = (a + b + c) / 2 
    area = sqrt(p * (p - a) * (p - b) * (p - c))  # площадь по формуле Герона
    return area
```
